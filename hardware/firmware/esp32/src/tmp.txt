#include <WiFi.h>
#include <HTTPClient.h>
#include <WebServer.h>
#include <EEPROM.h>
#include <ArduinoJson.h>
#include <main.h>

// Настройки EEPROM
#define EEPROM_SIZE 512

// Структура для хранения настроек
struct Settings
{
  char ssid[32];
  char password[64];
  char serverIp[16];
  int serverPort;
  char deviceName[32];
  bool configured;
};

Settings settings;
WebServer server(80);

// Режимы работы
bool isConfigMode = false;
unsigned long lastAttempt = 0;
const unsigned long configTimeout = 300000; // 5 минут

void setup()
{
  Serial.begin(115200);

  // Инициализация EEPROM
  EEPROM.begin(EEPROM_SIZE);
  loadSettings();

  // Проверяем, есть ли сохраненные настройки
  if (settings.configured)
  {
    Serial.println("Найдены сохраненные настройки. Пытаемся подключиться...");
    connectToWiFi();
  }
  else
  {
    Serial.println("Нет сохраненных настроек. Запускаем режим настройки...");
    startConfigMode();
  }
}

void loop()
{
  if (isConfigMode)
  {
    server.handleClient();

    // Автоматически выходим из режима настройки через 5 минут
    if (millis() - lastAttempt > configTimeout)
    {
      Serial.println("Таймаут режима настройки. Перезагрузка...");
      ESP.restart();
    }
  }
  else
  {
    // Основной режим работы
    if (WiFi.status() == WL_CONNECTED)
    {
      // Здесь ваш основной код
      sendData();
    }
    else
    {
      Serial.println("Потеряно соединение с WiFi. Пытаемся переподключиться...");
      connectToWiFi();
    }
    delay(10000); // Отправляем данные каждые 10 секунд
  }
}

void connectToWiFi()
{
  Serial.print("Подключение к ");
  Serial.println(settings.ssid);

  WiFi.mode(WIFI_STA);
  WiFi.begin(settings.ssid, settings.password);

  int attempts = 0;
  while (WiFi.status() != WL_CONNECTED && attempts < 20)
  {
    delay(500);
    Serial.print(".");
    attempts++;
  }

  if (WiFi.status() == WL_CONNECTED)
  {
    Serial.println("\nПодключено успешно!");
    Serial.print("IP адрес: ");
    Serial.println(WiFi.localIP());
    isConfigMode = false;
  }
  else
  {
    Serial.println("\nНе удалось подключиться. Запускаем режим настройки...");
    startConfigMode();
  }
}

void startConfigMode()
{
  isConfigMode = true;
  lastAttempt = millis();

  // Создаем точку доступа
  WiFi.mode(WIFI_AP);
  WiFi.softAP("ESP32_Config", "12345678");

  IPAddress apIP = WiFi.softAPIP();
  Serial.print("Точка доступа создана. IP: ");
  Serial.println(apIP);
  Serial.println("Подключитесь к WiFi 'ESP32_Config' пароль '12345678'");
  Serial.println("Откройте браузер и перейдите по адресу: http://192.168.4.1");

  // Настройка маршрутов веб-сервера
  server.on("/", handleRoot);
  server.on("/save", HTTP_POST, handleSave);
  server.on("/scan", handleScan);
  server.on("/status", handleStatus);

  server.begin();
}

void handleRoot()
{
  String html = R"rawliteral(
<!DOCTYPE html>
<html>
<head>
    <meta name="viewport" content="width=device-width, initial-scale=1">
    <meta charset="UTF-8">
    <title>Настройка ESP32</title>
    <style>
        body { font-family: Arial; text-align: center; margin-top: 50px; }
        input { width: 80%; padding: 10px; margin: 10px; }
        button { padding: 15px 30px; background: #4CAF50; color: white; border: none; cursor: pointer; }
        select { width: 80%; padding: 10px; margin: 10px; }
    </style>
</head>
<body>
    <h2>Настройка ESP32</h2>
    <form action="/save" method="post">
        <select id="ssid" name="ssid" required>
            <option value="">Выберите WiFi сеть</option>
        </select><br>
        <input type="password" name="password" placeholder="Пароль WiFi" required><br>
        <input type="text" name="serverIp" placeholder="IP сервера (например: 192.168.1.151)" required><br>
        <input type="number" name="serverPort" placeholder="Порт сервера" value="8080" required><br>
        <input type="text" name="deviceName" placeholder="Имя устройства" value="ESP32_1" required><br>
        <button type="submit">Сохранить и подключиться</button>
    </form>
    
    <script>
        // Сканирование WiFi сетей
        fetch('/scan')
            .then(response => response.json())
            .then(data => {
                const select = document.getElementById('ssid');
                data.networks.forEach(network => {
                    const option = document.createElement('option');
                    option.value = network;
                    option.text = network;
                    select.appendChild(option);
                });
            });
    </script>
</body>
</html>
)rawliteral";

  server.send(200, "text/html", html);
}

void handleScan()
{
  int n = WiFi.scanNetworks();
  String json = "{\"networks\":[";

  for (int i = 0; i < n; i++)
  {
    if (i > 0)
      json += ",";
    json += "\"" + WiFi.SSID(i) + "\"";
  }

  json += "]}";
  server.send(200, "application/json", json);
}

void handleSave()
{
  if (server.hasArg("ssid") && server.hasArg("password") &&
      server.hasArg("serverIp") && server.hasArg("serverPort") &&
      server.hasArg("deviceName"))
  {

    // Сохраняем настройки
    strlcpy(settings.ssid, server.arg("ssid").c_str(), sizeof(settings.ssid));
    strlcpy(settings.password, server.arg("password").c_str(), sizeof(settings.password));
    strlcpy(settings.serverIp, server.arg("serverIp").c_str(), sizeof(settings.serverIp));
    settings.serverPort = server.arg("serverPort").toInt();
    strlcpy(settings.deviceName, server.arg("deviceName").c_str(), sizeof(settings.deviceName));
    settings.configured = true;

    saveSettings();

    String html = "<!DOCTYPE html><html><head><meta charset='UTF-8'></head><body>";
    html += "<h2>Настройки сохранены!</h2>";
    html += "<p>ESP32 перезагружается и подключается к сети...</p>";
    html += "</body></html>";

    server.send(200, "text/html", html);
    delay(2000);
    ESP.restart();
  }
  else
  {
    server.send(400, "text/plain", "Missing parameters");
  }
}

void handleStatus()
{
  String status = "{\"status\":\"";
  if (WiFi.status() == WL_CONNECTED)
  {
    status += "connected\",\"ip\":\"" + WiFi.localIP().toString() + "\"";
  }
  else
  {
    status += "disconnected\"";
  }
  status += "}";
  server.send(200, "application/json", status);
}

void loadSettings()
{
  EEPROM.get(0, settings);
  if (settings.configured != true && settings.configured != false)
  {
    // Если память не инициализирована
    settings.configured = false;
  }
}

void saveSettings()
{
  EEPROM.put(0, settings);
  EEPROM.commit();
}

void sendData()
{
  if (WiFi.status() != WL_CONNECTED)
    return;

  HTTPClient http;

  String url = "http://" + String(settings.serverIp) + ":" +
               String(settings.serverPort) + "/api/data";

  http.begin(url);
  http.addHeader("Content-Type", "application/json");

  // Создаем JSON с данными
  String jsonData = "{";
  jsonData += "\"device\":\"" + String(settings.deviceName) + "\",";
  jsonData += "\"temperature\":" + String(random(20, 30)) + ",";
  jsonData += "\"humidity\":" + String(random(40, 60)) + ",";
  jsonData += "\"rssi\":" + String(WiFi.RSSI()) + ",";
  jsonData += "\"timestamp\":" + String(millis());
  jsonData += "}";

  int httpResponseCode = http.POST(jsonData);
 
  if (httpResponseCode > 0)
  {
    String response = http.getString();
    Serial.println("Данные отправлены. Ответ: " + response);
  }
  else
  {
    Serial.print("Ошибка отправки: ");
    Serial.println(httpResponseCode);
  }

  http.end();
}
